    Activité 1 – Les mathématiques du temps

    Situation

    Considérez le problème suivant : Écrivez un programme qui convertit un nombre de secondes en nombre d’heures, de minutes et de secondes. La méthode qui permet de lire le nombre de secondes est déjà codée. On vous propose trois implémentations de l’algorithme de conversion. Lisez les implémentations puis répondez aux questions .

    Exemples d’exécution

    1) Entrée : 10 898          Sortie : 3h 1m 38s
2) Entrée :  1 796          Sortie : 0h 29m 56s

    ///<summary>
/// Demande à la personne utilisatrice du système de saisir un nombre  
/// entier de secondes au clavier. Si le nombre est de format invalide ou  
/// négatif, alors elle doit recommencer la saisie.
///<returns>Le nombre de secondes saisies par la personne 
///utilisatrice.</returns> 
///</summary>
public static LireSecondes() {
    Console.WriteLine("Saisissez le nombre de secondes (nombre entier) " +
        "a convertir.");
    int secondes = -1;
    bool saisieValide = false;

    while(!saisieValide)
    {
        bool estEntier = int.TryParse(Console.ReadLine(), out secondes)
        if(!estEntier) 
        {
            Console.WriteLine("Le nombre saisie ne correspond pas au " + 
                "format entier. Saisissez la valeur de nouveau.");
        } 
        else
        {
            if(secondes < 0)
            {
                Console.WriteLine("Le nombre de seconde doit être plus " +
                    "grand ou égal à 0. Saisissez la valeur de nouveau.");
            }
            else
            {
                saisieValide = true;
            }
        }
    }

    return secondes;
}

    /**
 * Demande à la personne utilisatrice du système de saisir un nombre  
 * entier de secondes au clavier. Si le nombre est de format invalide ou  
 * négatif, alors elle doit recommencer la saisie.
 *
 * Retour : le nombre de secondes saisies par la personne utilisatrice.
 */ 
function lireSecondes() {
    let secondes = 0;
    let saisieValide = false;

    while (!saisieValide) {
        secondes = parseInt(prompt("Saisissez le nombre de secondes " +
            "(nombre entier) à convertir."));

        if (isNaN(secondes) || !Number.isInteger(secondes)) {
            alert("Le nombre saisi ne correspond pas au format entier. " +
                "Saisissez la valeur à nouveau.");
        } else {
            if (secondes < 0) {
                alert("Le nombre de seconde doit être plus grand ou " +
                    "égal à 0. Saisissez la valeur à nouveau.");
            } else {
                saisieValide = true;
            }
        }
    }

    return secondes;
}

    public static void  ConvertirSecondes() {
    int secondesEntrees = LireSecondes();

    for(int heures = 0; secondesEntrees >= 3600; secondesEntrees -= 3600) {
        heures++;
    }
    for(int minutes = 0; secondesEntrees >= 60; secondesEntrees -= 60) {
        minutes++;
    }
    Console.WriteLine($"{heures}h {minutes}m {secondesEntrees}s");
}

    int minutes = tempsConvertir / 60;
