    Gestion d'actions multiples

    Contenus de la page

    4-3 Accéder à la boucle de jeu pour les mises à jour du contrôle de la simulation

    Temps requis

    15 minutes

    La gestion d'une réaction sur plusieurs objets simultanément peut poser une problème lors de la conception du jeu. Prenons l'exemple d'un jeu où un joueur doit fuir un monstre et trouver un objet dans un certain temps donné, autrement il perd la partie. Un ControleurJeu est responsable de gérer la logique du jeu.

    ///<summary>
///Objet responsabele de la gestion du jeu. Il vérifie si la partie et terminée et réagit adéquatement.
///<summary>
public class ControleurJeu : MonoBehavior 
{
    [SerializeField, Tooltip("Le joueur principal")]
    private Joueur joueur;

    [SerializeField, Tooltip("L'ennemi qui poursuit le joueur")]
    private Ennemi ennemi;

    [SerializeField, Tooltip("L'élément de UI qui affiche le temps qui reste au joueur.")]
    private CompteurTemps compteur;

    // Variable qui compte le temps passer à rechercher
    private float tempsRecherche = 0.0f;

    [SerializeField, Tooltip("La fenetre de UI qui affiche les options lorsque la partie est gagnée")]
    private GameObject fenetreVictoire;

    [SerializeField, Tooltip("La fenetre de UI qui affiche les options lorsque la partie est perdue")]
    private GameObject fenetreDefaite;

    // Indique si l'objet recherché à été trouvé
    private bool ObjetTrouve {get; set;}

    private void Update() 
    {
        if(!ObjetTrouve && tempsRecherche < 60.0f) {
            tempsRecherche += Time.deltaTime;
            compteur.AfficherTempsRestant(60.0f - tempsRecherche);
        }
        if(ObjetTrouve)
        {
            // Bloc d'action lorsque victoire
            joueur.PeutBouger = false;
            joueur.gameObject.GetComponent<Animator>().SetTrigger("Victoire");
            ennemi.vitesse = 0;
            ennemi.gameObject.GetComponent<Animator>().SetTrigger("Défaite");
            compteur.gameObject.SetActive(false);
            fenetreVictoire.SetActive(true);
        }
        if(tempsRecherche >= 60.0f) {
            // Bloc d'action lorsque défaite
            joueur.PeutBouger = false;
            joueur.gameObject.GetComponent<Animator>().SetTrigger("Défaite");
            ennemi.vitesse = 0;
            ennemi.gameObject.GetComponent<Animator>().SetTrigger("Victoire");
            compteur.gameObject.SetActive(false);
            fenetreDefaite.SetActive(true);
        }
    }
}

    Ce code comporte plusieurs problèmes dans sa conception :

    1. On vérifie à chaque frame si l'objet est trouvé ou non;
    2. On test toujours si l'on doit ajouter du temps au temps écoulé;
    3. Les deux blocs d'actions :
    4. Possèdes des actions communes;
    5. Appellent plusieurs fois des méthodes sur les mêmes objets.

    L'idée principale va être d'encapsuler les opérations d'un objet dans ce dernier. Également, on va éviter de répéter les lignes de code même au sein des classes après l'encapsulation des éléments.

///<summary>
///Contrôle les éléments de l'interface utilisateur.
///</summary>
public class ControleurUI : MonoBehaviour {

    [SerializeField, Tooltip("La fenetre de UI qui affiche les options lorsque la partie est gagnée")]
    private GameObject fenetreVictoire;

    [SerializeField, Tooltip("La fenetre de UI qui affiche les options lorsque la partie est perdue")]
    private GameObject fenetreDefaite;

    [SerializeField, Tooltip("L'élément de UI qui affiche le temps qui reste au joueur.")]
    private CompteurTemps compteur;

    ///<summary>
    ///Affiche la fenêtre de la fin de la partie.
    ///<param name="victoire">Indique si le joueur a gagné.</param>
    ///</summary>
    public void AfficherFenetreFin(bool victoire)  
    {
        GameObject fenetre = victoire ? fenetreVictoire : fenetreDefaite
        fenetre.SetActive(true);
        compteur.gameObject.SetActive(false);
    }

    ///<summary>
    ///Affiche la valeur du temps restant au joueur avant la fin de la partie.
    ///<param name="victoire">La quantité de temps restant.</param>
    ///</summary>
    public void AfficherTempsRestant(float tempsRestant)
    {
        compteur.SetText(tempsRestant);
    }
}

///<summary>
///Objet responsable de la gestion du jeu. Il vérifie si la partie et terminée et réagit adéquatement.
///<summary>
public class ControleurJeu : MonoBehavior 
{
    [SerializeField, Tooltip("Le joueur principal")]
    private Joueur joueur;

    [SerializeField, Tooltip("L'ennemi qui poursuit le joueur")]
    private Ennemi ennemi;

    [SerializeField, Tooltip("Le controleur du UI.")]
    private ControleurUI controleurUI;

    // Variable qui compte le temps qui reste
    private float tempsRestant;

    [SerializeField, Tooltip("Temps disponible pour trouver l'objet")]
    private float tempsDisponible;

    // Indique si l'objet recherché à été trouvé
    public bool ObjetTrouve {get; set;}    

    private void Start()
    {
        StartCoroutine(nameof(TempsRecherche));   
        tempsRestant = tempsDisponible;
    }

    ///<summary>
    ///Réalise les opérations de fin de partie pour tous les objets contrôlés.
    ///<param name="victoire">Indique si le joueur a gagné la partie.
    ///</summary>
    public void ActionFinPartie(bool victoire) {
        joueur.ActionFinPartie(victoire);
        ennemi.ActionFinPartie(victoire);
        controleurUI.AfficherFenetreFin(vicgtoire)
    }

    ///<summary>
    ///Coroutine qui gère l'écoulement du temps dans la recherche des objets.
    ///</summary>    
    private IEnumerator TempsRecherche()
    {
        while(!ObjetTrouve && tempsRestant > 0.0f) {
            tempsRestant -= Time.deltaTime;
            controleurUI.AfficherTempsRestant(tempsRestant);
        }

        ActionFinPartie(ObjetTrouve);
    }
}

    Parmis les modifications on retrouve les suivantes :

    • On pense en termes de temps restant parce que c'est que l'on affiche à l'utilisateur et on utilise une variable exposée pour contrôler sa valeur initiale.
    • Les objets Joueur et Ennemi sont responsables de définir les actions de fin de partie qu'ils font. Si l'on les change (exemple destruction, son, effet visuel supplémentaire...), on peut le faire sans changer le ContrôleurJeu.
    • Les objets de UI sont gérés dans leur propre contrôleur (ce ne sont pas des objets de jeu à proprement parler).
    • Les actions de fin de partie sont regroupées dans une méthode et un booléen gère le type de partie.
    • Une coroutine gère le temps de recherche et peut être arrêtée en fin de partie.
    • On arrête la coroutine par elle-même (la plupart des utilisations de StopCoroutine peuvent être évitées en utilisant un design adéquat du code). Si l'objet a été trouvé, alors ObjetTrouve est true et on lance les actions de fin avec le paramètre victoire à true, si le délai a expiré, alors ObjetTrouve est false, et on lance la fin avec une victoire à false.

    Le problème du couplage

    Le contrôleur de jeu voit tous les éléments, ce qui peut devenir un problème si le jeu gère beaucoup d'élément. On peut utiliser un événement pour gérer la situation. Chaque objet est responsable d'écouter les événements onFinPartie et onChangementTempsDisponible s'il doit exécuter des actions à ce moment.

    ///<summary>
///Objet responsable de la gestion du jeu. Il vérifie si la partie et terminée et réagit adéquatement.
///<summary>
public class ControleurJeu : MonoBehavior 
{
    // Indique que la partie est terminée
    public UnityEvent<bool> onFinPartie;                // bool => victoire

    // Indique que le temps restant a diminué
    public UnityEvent<float> onChangementTempsRestant;   // float => valeur de temps restant

    // Variable qui compte le temps qui reste
    private float tempsRestant;

    [SerializeField, Tooltip("Temps disponible pour trouver l'objet")]
    private float tempsDisponible;    

    // Indique si l'objet recherché à été trouvé
    public bool ObjetTrouve {get; set;}

    private void Start()
    {
        StartCoroutine(nameof(TempsRecherche));   
        tempsRestant = tempsDisponible;
    }

     ///<summary>
    ///Coroutine qui gère l'écoulement du temps dans la recherche des objets.
    ///</summary>    
    private IEnumerator TempsRecherche()
    {
        while(!ObjetTrouve && tempsRestant > 0.0f) {
            tempsRestant -= Time.deltaTime;
            onChangementTempsRestant?.Invoke(tempsRestant);
        }

        onFinPartie?.Invoke();
    }
}

    Avec les événements, le contrôleur de jeu ne contient plus que la logique du jeu (soit qu'il faut trouver un objet dans un temps imparti). Souvent un contrôleur long relativement à la complexité du jeu est un signe de problèmes dans le design du jeu.
